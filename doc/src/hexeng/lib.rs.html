//! hexeng: a crate for handling hexagonal tilings / grids / tesselation.
//!
//! Provides calculations on hexagonal tiles, using
//! [nalgebra](http://www.nalgebra.org) for underlying linear algebra and
//! data structures.
//!
//! # Coordinate frames
//!
//! The core concept of this crate is the coordinate frame, represented by the
//! [Frame](trait.Frame.html) trait. There are three coordinate frames:
//!
//! * Axial coordinates - pairs of integers which uniquely identify a hexagon
//!   on the grid.
//! * World coordinates - a pair of floats representing a point in a continuous
//!   space.
//! * View coordinates - an arbitrary non-singular matrix transfomation on the
//!   world coordinates.
//!
//! It is possible to convert coordinates between a pair of frames. The
//! conversion from Axial to World determines a 2D Euclidean location for each
//! hexagon on the grid. The conversion from World to View is intended to allow
//! the grid to be displayed with arbitrary scale/rotation (etc) on a display.
//!

//! The choice of geometry in the World frame is to make the conversion
//! convenient. An explanation of the full conversion algorithm can be found
//! [here](https://jthorniley.github.io/notebooks/hex.html).
//!
//! Converting from a World coordinate to Axial coordinates will find the
//! hexagon that contains that world coordinate. Converting from the Axial
//! coordinate back to World coordinates will find the _origin_ of the
//! specified hexagon, i.e. the `0, 0` point on the geometry shown below:
//!
//! <div>
//! <img src="https://jthorniley.github.io/notebooks/_images/hex_5_0.svg"/>
//! </div>
//!
//! This means that the point retrieved by the Axial to World conversion is
//! *not* inside the hexagon at that axial address. The following example
//! may clarify this further:
//!
//! ```
//! use hexeng::{World, Axial, Recast, ViewTransform};
//! use nalgebra::Matrix3;
//!
//! // To use the Recast trait, we must define the View transform (but
//! // since we are not using the View frame in this example, its value is
//! // not important - just use the identity matrix).
//! let view_transform = Matrix3::identity();
//!
//! // World point 1.5, 6.0 is inside the 1, 3 hexagon
//! let world = World::from([1.5, 6.0]);
//! let axial: Axial = world.recast(&view_transform);
//! assert_eq!(axial.as_slice(), &[1, 3]);
//!
//! // The "origin coordinate" of the 1, 3 hexagon is found
//! // by converting back to world coordinates. Note that the
//! // origin is actually *not* inside the 1, 3 hexagon as it is
//! // at the lower left corner!
//! let origin: World = axial.recast(&view_transform);
//! assert_eq!(origin.as_slice(), &[1.0, 5.0]);
//! assert_eq!(Recast::<Axial>::recast(origin, &view_transform).as_array(), [0, 2]);
//! ```
//!
//! ## World to/from View coordinates
//!
//! The conversion between world and view is defined by a transformation
//! that must be supplied to the [recast](trait.Recast.html#tymethod.recast)
//! method by way of the [ViewTransform](trait.ViewTransform.html) trait.
//!
//! The transformation should be specified using the geometry system provided
//! by the [nalgebra](https://www.nalgebra.org/rustdoc/nalgebra/geometry/index.html)
//! crate.
//!
//! ```
//! use hexeng::{World, View, Recast, ViewTransform};
//! use nalgebra::{Vector2, Matrix3};
//!
//! fn get_transform(translation: [f64; 2], scale: [f64; 2]) -> Matrix3<f64> {
//!     Matrix3::identity()
//!         .append_translation(&Vector2::from(translation))
//!         .append_nonuniform_scaling(&Vector2::from(scale))
//! }
//!
//! let world = World::from([2.0, 3.5]);
//! let view: View = world.recast(&get_transform([0.1, 0.2], [1., 1.]));
//! assert_eq!(view.as_slice(), &[2.1, 3.7]);
//!
//! // Converting from view to world applies the inverse transform
//! let view = View::from([2.0, 3.5]);
//! let world: World = view.recast(&get_transform([0., 0.], [0.5, 0.5]));
//! assert_eq!(world.as_slice(), &[4.0, 7.0]);
//! ```
mod convert;

use nalgebra::{Matrix3, Point2, Projective2, Scalar};
use num_traits::{One, Zero};
use std::convert::From;
use std::marker::PhantomData;
use std::ops::{Deref, Div, DivAssign};

/// Implemented by objects that can be treated as transformations between
/// world/view coordinates.
pub trait ViewTransform: Sized {
    /// Reference type for the view transform
    ///
    /// This can be customised by the implementation to use anything that
    /// dereferences to a `Projective2<f64>` - for example a normal reference
    /// or a `Box`.
    type Ref: Deref<Target = Projective2<f64>>;

    /// Retrieve the view transform associated with this instance.
    fn view_transform(&self) -> Self::Ref;
}

/// Treat a Matrix3 as a ViewTransform.
///
/// For convenience, the easiest way to specify a
/// [ViewTransform](trait.ViewTransform.html) to the
/// [Recast](trait.Recast.html) trait is to simply used a 3x3 matrix
/// defining the transform.
///
/// *Important* The matrix must be invertible! This will not be checked (the
/// client must ensure this).
impl ViewTransform for Matrix3<f64> {
    type Ref = Box<Projective2<f64>>;

    fn view_transform(&self) -> Self::Ref {
        Box::new(Projective2::from_matrix_unchecked(self.clone()))
    }
}

/// Used to identify Frame objects that are in Axial coordinates by typing.
#[derive(PartialEq, Clone)]
#[cfg_attr(test, derive(Debug))]
pub enum TAxial {}

/// Used to identify Frame objects that are in World coordinates by typing.
#[derive(PartialEq, Clone)]
#[cfg_attr(test, derive(Debug))]
pub enum TWorld {}

/// Used to identify Frame objects that are in View coordinates by typing.
#[derive(PartialEq, Clone)]
#[cfg_attr(test, derive(Debug))]
pub enum TView {}

/// An object representing a point within a specified frame.
///
/// The frame is included in the type by the parameter T, which
/// should be one of TAxial, TWorld or TView.
#[derive(PartialEq, Clone)]
#[cfg_attr(test, derive(Debug))]
pub struct Frame<V: Scalar, T>(Point2<V>, PhantomData<T>);

impl<V: Scalar, T> Frame<V, T> {
    pub fn as_slice(&self) -> &[V] {
        self.0.coords.as_slice()
    }
    pub fn as_array(self) -> [V; 2] {
        self.0.coords.into()
    }
}

/// Axial coordinates
pub type Axial = Frame<i64, TAxial>;
/// World coordinates
pub type World = Frame<f64, TWorld>;
/// View coordinates
pub type View = Frame<f64, TView>;

impl<V: Scalar + One, T> From<Point2<V>> for Frame<V, T> {
    fn from(pt: Point2<V>) -> Frame<V, T> {
        Frame(pt, PhantomData)
    }
}

impl<V: Scalar + One + Zero + Div<Output = V> + DivAssign, T> From<Frame<V, T>> for Point2<V> {
    fn from(fr: Frame<V, T>) -> Point2<V> {
        Point2::from(fr.0.coords)
    }
}

impl<V: Scalar + One, T> From<[V; 2]> for Frame<V, T> {
    fn from(pt: [V; 2]) -> Frame<V, T> {
        Frame(Point2::from(pt), PhantomData)
    }
}

/// Trait implemented by all the frame types (Axial, View and World).
///
/// This can be used as a base type that can accept any Frame object.
///
/// # Example
///
/// ```
/// use hexeng::{Recast, ViewTransform, AnyFrame, View, World};
/// use nalgebra::Matrix3;
///
/// fn check_value(value: impl hexeng::AnyFrame) {
///     let view: View = value.recast(&Matrix3::identity());
///     assert_eq!(view.as_slice(), &[1., 1.])
/// }
///
/// check_value(World::from([1., 1.]));
/// check_value(View::from([1., 1.]));
/// ```
///
pub trait AnyFrame: Clone + Recast<World> + Recast<View> + Recast<Axial> {}

impl AnyFrame for Axial {}
impl AnyFrame for World {}
impl AnyFrame for View {}

/// Implemented by Frame objects, to allow converting to other frames
pub trait Recast<F: AnyFrame>: Sized {
    /// Consume the current object and convert it to another frame.
    fn recast(self, _view_state: &impl ViewTransform) -> F;
}

/// Convert Axial frame to World
impl Recast<World> for Axial {
    fn recast(self, _view_state: &impl ViewTransform) -> World {
        World::from(convert::axial_to_world(self.into()))
    }
}

/// Convert World frame to Axial
impl Recast<Axial> for World {
    fn recast(self, _view_state: &impl ViewTransform) -> Axial {
        Axial::from(convert::world_to_axial(self.into()))
    }
}

/// Convert World frame to View
impl Recast<World> for View {
    fn recast(self, view_state: &impl ViewTransform) -> World {
        let t = view_state.view_transform().inverse();
        World::from(t.transform_point(&self.0))
    }
}

/// Convert View frame to World
impl Recast<View> for World {
    fn recast(self, view_state: &impl ViewTransform) -> View {
        let t = view_state.view_transform();
        View::from(t.transform_point(&self.0))
    }
}

/// Convert from View to Axial (goes via conversion to World)
impl Recast<View> for Axial {
    fn recast(self, view_state: &impl ViewTransform) -> View {
        let world: World = self.recast(view_state);
        world.recast(view_state)
    }
}

/// Convert from Axial to View (goes via conversion to World)
impl Recast<Axial> for View {
    fn recast(self, view_state: &impl ViewTransform) -> Axial {
        let world: World = self.recast(view_state);
        world.recast(view_state)
    }
}

/// Implement no-op where recast is called to the same reference frame
impl<F: AnyFrame> Recast<F> for F {
    fn recast(self, _view_state: &impl ViewTransform) -> F {
        self
    }
}
